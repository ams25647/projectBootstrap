My Javascript example for my ITP 140


          const clock = document.getElementById('clock');
          const date = document.getElementById('date');
          const formatSelect = document.getElementById('formatSelect');
        
          setInterval(() => {
            const now = new Date();
            let hours = now.getHours();
            const minutes = now.getMinutes();
            const seconds = now.getSeconds();
            const amPm = hours < 12 ? 'AM' : 'PM';
        
            // Convert to 12-hour format if selected
            if (formatSelect.value == "12") {
              hours = hours % 12;
              hours = hours ? hours : 12;
            }
        
            hours = ('0' + hours).slice(-2);
            const minutesPadded = ('0' + minutes).slice(-2);
            const secondsPadded = ('0' + seconds).slice(-2);
        
            clock.textContent = `${hours}:${minutesPadded}:${secondsPadded} ${formatSelect.value == "12" ? amPm : ""}`;
            date.textContent = now.toDateString();
          }, 1000);
